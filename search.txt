function search(problem) returns soloution:
	for depth from 0 to infinity:
		result = depth_limited_search(problem, dept)
		if result != cutoff:
			return result


function depth_limited_search(problem, limit) returns soloution or failure:
	return recursive_dls(problem.initial_state, problem, limit)


function recursive_dls(node, problem, limit) returns soloution or failure:
	if node is goal:
		 return soloution(node)

		return
